# Test notebook

Let's test simple code:

```zig
const std = @import("std");
const zig_ml = @import("zig_ml");
const tensor = zig_ml.tensor;
const nn = zig_ml.nn;
const autodiff = zig_ml.autodiff;
```

```zig
fn printTensorInfo(t: *tensor.Tensor) void {
    std.debug.print("Tensor shape: [", .{});
    for (0..t.shape.len) |idx| {
        std.debug.print("{}, ", .{t.shape.dims[idx]});
    }
    std.debug.print("], data: [", .{});
    const limit = if (t.data.len > 4) @as(usize, 4) else t.data.len;
    for (0..limit) |idx| {
        std.debug.print("{d:.4}, ", .{t.data[idx]});
    }
    std.debug.print("...]\n", .{});
}
```

Now let's test initializing an Embedding layer and running forward pass:

```zig
const allocator = init.arena.allocator();
var prng = std.Random.DefaultPrng.init(1337);
var emb = try nn.Embedding.init(allocator, 10, 4, prng.random());

var graph = autodiff.Graph.init(allocator);

const x_node = try graph.tensorND(&.{2, 3}, false);
x_node.data[0] = 0; x_node.data[1] = 1; x_node.data[2] = 2;
x_node.data[3] = 3; x_node.data[4] = 4; x_node.data[5] = 5;

const y = try emb.forward(allocator, &graph, x_node);
std.debug.print("Block 3 execution: forward completed\n", .{});
printTensorInfo(y);
```

Now let's run backward pass using the variables from Block 3:

```zig
// Let's set some gradients on y
@memset(y.grad, 1.0);

// Backward propagation
std.debug.print("Block 4 execution: running backward pass\n", .{});
try graph.backward(y);

std.debug.print("Embedding weights gradients:\n", .{});
for (0..6) |idx| {
    std.debug.print("Row {}: [", .{idx});
    for (0..4) |d| {
        std.debug.print("{d:.4}, ", .{emb.weight.grad[idx * 4 + d]});
    }
    std.debug.print("]\n", .{});
}
```
